% Create the \tikzoverlay function which allows the placing of TikZ nodes at
% arbitrary locations on the page using the (current page) node.  For example,
% for a node placed at the bottom right corner, one would use:
%
%   \tikzoverlay [anchor=south east] at (current page.south east) {My text};
%
% Note the final semicolon, it is required.
\newcommand\tikzoverlay{%
    \tikz[remember picture, overlay]\node
}

% This command places text in the bottom right corner, useful for the source of
% pictures for example.
\newcommand{\source}[1]{%
    \tikzoverlay [anchor=south east] at (current page.south east) {%
        \begin{beamercolorbox}[ht=0.5cm, right]{source}
            \usebeamerfont{source}\usebeamercolor[fg]{source} Source: {#1}
        \end{beamercolorbox}
    };
}

\section{Introduction}

\begin{frame}
    \frametitle{Introduction}

    Beamer allows you to create very nice slides.  Here is a fairly simple
    slideshow to demonstrate a few basic uses of Beamer.

    As you might expect from \LaTeX, it tries to positions everything \emph{just
        right}; having said that, \LaTeX was originally designed for print and
    presentation may require a bit more trial-and-error.
\end{frame}
